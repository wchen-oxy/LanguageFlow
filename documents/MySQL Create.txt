CREATE TABLE Bhutia(

ipa varchar(50),
category varchar (50),
eng_trans varchar (50),
bhut_rom_formal varchar (50),
bhut_rom_informal varchar (50),
bhut_script_formal varchar (50),
bhut_script_informal varchar (50),
example varchar (200),
spoken_b varchar (150),
spoken_e varchar (150),
PRIMARY KEY (eng_trans)
);
